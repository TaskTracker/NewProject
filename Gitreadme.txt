1. git init <folder Name>                 folder now is a repository
2. git add .                              add all files to staging
3. git commit -m "message"                snapshot
4. git status                             status of files
5. git log                                commit history
6. git diff                               view differences
7. git diff --cached                      differences in staging
8. git remote add origin                  adds remote repository
"name of remote repository" 
9. git push name  branchName               add to the master
10. git branch branchName                  create a new branch
